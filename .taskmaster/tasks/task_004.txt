```typescript
// vocabulary.service.ts
export const getVocabularyForDay = async (cycle: number, day: number, languagePair: string): Promise<VocabularyWord[]> => {
  const [sourceLanguage, targetLanguage] = languagePair.split('-');
  
  // 사이클과 날짜에 따른 단어 범위 계산
  const wordsPerDay = 25;
  let totalRange = 0;
  let startIndex = 0;
  
  switch(cycle) {
    case 1: // 하루 25단어, 20일
      totalRange = 25;
      startIndex = (day - 1) * 25;
      break;
    case 2: // 50단어 범위, 10일
      totalRange = 50;
      startIndex = (day - 1) * 50;
      break;
    // ... 다른 사이클들
  }
  
  // 범위 내 단어에 대해 Firestore 쿼리
  const q = query(
    collection(db, 'vocabularies'),
    orderBy('id'),
    startAt(startIndex),
    limit(totalRange)
  );
  
  const snapshot = await getDocs(q);
  return snapshot.docs.map(doc => doc.data() as VocabularyWord);
};
```
